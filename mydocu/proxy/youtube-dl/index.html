youtube-dl

installation

binary

sudo curl -L https://yt-dl.org/downloads/latest/youtube-dl -o /usr/local/bin/youtube-dl

sudo chmod a+rx /usr/local/bin/youtube-dl

docker

youtube-dl-webui pip list --outdated --format=columns pip install -U youtube-dl -i https://pypi.tuna.tsinghua.edu.cn/simple

usage

youtube-dl --verbose --proxy socks5://192.168.1.172:1080 -o 'a.mp4' https://www.youtube.com/watch?v=WUIVdiW_fAQ youtube-dl --verbose --proxy socks5://192.168.1.172:1080 -o 'a.f251.mp4' https://www.youtube.com/watch?v=WUIVdiW_fAQ youtube-dl --verbose --list-extractors

// 查看可用视频类型 youtube-dl --verbose --proxy socks5://192.168.1.172:1080 -F https://www.youtube.com/watch?v=WUIVdiW_fAQ youtube-dl --verbose --proxy socks5://192.168.1.172:1080 -f 249 -o 'a.f249.webm' https://www.youtube.com/watch?v=WUIVdiW_fAQ

youtube-dl -o '%(title)s by %(uploader)s on %(upload_date)s in %(playlist)s.%(ext)s' https://www.youtube.com/watch?v=iJvr0VPsn-s youtube-dl youtube-dl -a url.txt

// 仅下载音频，默认Ogg（opus)格式；转换mp3需ffmpeg或avconv youtube-dl -x --audio-format mp3 https://www.youtube.com/watch?v=iJvr0VPsn-s

youtube-dl --write-description --write-info-json --write-annotations --write-sub --write-thumbnail https://www.youtube.com/watch?v=iJvr0VPsn-s --proxy "socks5://127.0.0.1:1080/"

sample

youtube-dl --verbose --proxy socks5://127.0.0.1:1080 -F https://www.youtube.com/watch?v=Kg-Gj2qm7X8

youtube-dl --verbose --proxy http://192.168.1.172:8118 -F https://www.youtube.com/watch?v=sSII1ou7yKM youtube-dl --verbose --proxy socks5://127.0.0.1:1080 -f 600 https://www.youtube.com/watch?v=LdtU_xgzxKg

// Kubeflow Pipelines - Helping developers from prototyping to production (Google Cloud AI Huddle)--kBjoe1UoXE youtube-dl --verbose --proxy socks5://127.0.0.1:1080 -f 18 https://www.youtube.com/watch?v=-kBjoe1UoXE

external downloader

alias ytdl='youtube-dl --verbose --proxy socks5://192.168.1.172:1080' youtube-dl --verbose \ --proxy 192.168.1.172:8118 \ -f 18 \ --external-downloader aria2c \ --external-downloader-args "-x 16 -k 1M" \ --write-auto-sub --sub-lang en --embed-sub \ https://www.youtube.com/watch?v=I6iMznIYwM8

alias ytdl249='youtube-dl --verbose \ --proxy 192.168.1.172:8118 \ -f 249 \ --external-downloader aria2c \ --external-downloader-args "-x 16 -k 1M"'

ytdl-aria \ -f 160+249 \ --write-auto-sub \ --sub-lang en \ --embed-sub \ https://www.youtube.com/watch?v=f6SOu9cq4OU

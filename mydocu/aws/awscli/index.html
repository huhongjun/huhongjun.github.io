aws-cli

官方参考

1、python环境准备

conda create -n awscli  pip
conda activate awscli
d:\Anaconda3\Scripts\activate awscli

2、awscli安装

activate awscli
pip install awscli -i https://pypi.tuna.tsinghua.edu.cn/simple
aws --version
aws configure

ec2

aws ec2 describe-regions
aws ec2 describe-instances --filters "Name=instance-type,Values=t3.nano" --query "Reservations[].Instances[].InstanceId"
aws ec2 terminate-instances --instance-ids i-006d3959680f4c412    
aws ec2 stop-instances --instance-ids i-006d3959680f4c412
aws ec2 stop-instances --instance-ids i-0e5343e2d46385c3c

安全组

//安全组添加端口开放    IPv4: 0.0.0.0/0 , 203.0.113.0/24
aws ec2 authorize-security-group-ingress --group-id sg-45f6b12c --protocol tcp --port 10000 --cidr 0.0.0.0/0

aws ec2 describe-security-groups --group-ids sg-45f6b12c

S3

aws s3 ls
aws s3 ls s3://bucket [--region us-west-2]

//删除指定的S3 bucket和其中的所有内容
aws s3 rb s3://bucket-name --force
递归地复制PC中的一个目录以及其子文件到Amazon S3
aws s3 cp MyFolder s3://bucket-name -- recursive [--region us-west-2]

launch-template

aws ec2 delete-launch-template --launch-template-id lt-0d95007be92bf3160 --region us-east-2

JMESPath帮助

-- query 使用JMESPath（一种JSON查询语言）从结果中提取数据

aws ec2 describe-instances --query 'Reservations[*].Instances[*].[InstanceId,  Placement.AvailabilityZone, State.Name,  InstanceType, PublicIpAddress, PrivateIpAddress, Tags[0].Value, Tags[1].Value]' --output text

aws iam list-users --query 'Users[*].[UserName,CreateDate,UserId]' --output table
